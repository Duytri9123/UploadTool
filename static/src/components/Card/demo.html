Card Component Demo

Basic Cards

new Card({ body: 'Simple card with only body content' });
new Card({ header: 'Card Title', body: 'Card with header and body' });
new Card({ header: 'Full Card', body: 'Card content', footer: 'Card footer' });

Card Variants

new Card({ header: 'Default', body: 'Default variant', variant: 'default' });
new Card({ header: 'Bordered', body: 'Bordered variant', variant: 'bordered' });
new Card({ header: 'Elevated', body: 'Elevated variant', variant: 'elevated' });

Collapsible Cards

new Card({
  header: 'Collapsible Card',
  body: 'Click the header to toggle',
  collapsible: true,
  collapsed: false
});

Cards with Rich Content

// Cards can contain HTML elements
const bodyElement = document.createElement('div');
bodyElement.innerHTML = '<p>Rich content</p>';
new Card({ header: 'Rich Card', body: bodyElement });

Dynamic Updates

card.setHeader('New Header');
card.setBody('New Body');
card.setFooter('New Footer');
card.toggle(); // Toggle collapsed state
card.expand(); // Expand card
card.collapse(); // Collapse card
