Usage Example

import { Card } from './components/Card/Card.js'; const card = new Card({ title: 'My Card', collapsible: true, collapsed: false, bodyContent: '<p>Card content here</p>', footerContent: '<button>Save</button>', onToggle: (collapsed) => console.log('Collapsed:', collapsed) }); document.body.appendChild(card.element);
